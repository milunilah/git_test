### Install Syncthing
% sudo mkdir -p /etc/apt/keyrings   # create folder for release keys
% sudo curl -L -o /etc/apt/keyrings/syncthing-archive-keyring.gpg https://syncthing.net/release-key.gpg   # provide release key
% echo "deb [signed-by=/etc/apt/keyrings/syncthing-archive-keyring.gpg] https://apt.syncthing.net/ syncthing stable-v2" | sudo tee /etc/apt/sources.list.d/syncthing.list   # Add the "stable-v2" channel to your APT sources
% echo "deb [signed-by=/etc/apt/keyrings/syncthing-archive-keyring.gpg] https://apt.syncthing.net/ syncthing candidate" | sudo tee /etc/apt/sources.list.d/syncthing.list   # Add the "candidate" channel to your APT sources
% printf "Package: *\nPin: origin apt.syncthing.net\nPin-Priority: 990\n" | sudo tee /etc/apt/preferences.d/syncthing.pref   # Increase preference of Syncthing's packages ("pinning")

### Download VSCode

sudo apt update

### Install Git
sudo add-apt-repository ppa:git-core/ppa
sudo apt update
sudo apt install git
git --version   # verify version (at least 2.28)

### Setup Git
git config --global user.name "Your Name"
git config --global user.email yourname@example.com
git config --global init.defaultBranch main   # to set main folder
git config --global pull.rebase false   # set default branch reconciliation behavior to merging
git config --global core.editor "code --wait"   # to not forget commit message
git config --get user.name   # verify
git config --get user.email   # verify

### Setup SSH Key -> Identifier for machine to connect to the server without having to type password everytime
ls ~/.ssh/id_ed25519.pub   # check if you already have a SSH key
ssh-keygen -t ed25519   # generates a key pair -> save in password manager
